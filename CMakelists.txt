cmake_minimum_required(VERSION 3.22.1)

project(IOTA) #project name

set(BUILD_SHARED_LIBS ON)
set(SRC ${PROJECT_SOURCE_DIR}/src/)

# find_package(Freetype REQUIRED)

add_executable(${PROJECT_NAME}
# CMAKE CPP INCLUDE
main.cpp
)
# find_package(glfw3 REQUIRED) #assuming you installed it somwhere
target_include_directories(
    ${PROJECT_NAME}
    PRIVATE ${PROJECT_SOURCE_DIR}/hed
    PRIVATE ${PROJECT_SOURCE_DIR}/Library/SFML/include
)
if(${CMAKE_SYSTEM_NAME} MATCHES "Darwin")
	message("You're on Mac OS !")
elseif(${CMAKE_SYSTEM_NAME} MATCHES "Linux")

target_link_libraries(${PROJECT_NAME} 
        ${PROJECT_SOURCE_DIR}/Library/SFML/lib/libsfml-audio.a
        ${PROJECT_SOURCE_DIR}/Library/SFML/lib/libsfml-graphics.a
        ${PROJECT_SOURCE_DIR}/Library/SFML/lib/libsfml-system.a
        ${PROJECT_SOURCE_DIR}/Library/SFML/lib/libsfml-window.a
        gdi32
        opengl32
    )
	message("You're on Linux, using libXXX.a!")
elseif(${CMAKE_SYSTEM_NAME} MATCHES "Windows")
set(CMAKE_WINDOWS_EXPORT_ALL_SYMBOLS ON)
# https://stackoverflow.com/questions/4702732/the-program-cant-start-because-libgcc-s-dw2-1-dll-is-missing
# gilgamesh smiles upon you hardmath
add_definitions("-DSFML_STATIC")
set(CMAKE_EXE_LINKER_FLAGS "-static -static-libgcc -static-libstdc++")

target_link_libraries(
    ${PROJECT_NAME} 
        ${PROJECT_SOURCE_DIR}/Library/SFML/lib/libsfml-graphics-s.a
        ${PROJECT_SOURCE_DIR}/Library/SFML/lib/libsfml-window-s.a
        ${PROJECT_SOURCE_DIR}/Library/SFML/lib/libsfml-system-s.a
        winmm
        gdi32
        opengl32
        ${PROJECT_SOURCE_DIR}/Library/SFML/lib/libfreetype.a
    )
	message("You're on Windows !")
endif(${CMAKE_SYSTEM_NAME} MATCHES "Darwin")
